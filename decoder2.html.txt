    <script>
        /**
         * Карта посимвольного маппинга (Векторный декодер латыни).
         * Сюда вы забиваете ID ваших векторных символов или их цифровые коды.
         * Код сопоставляет их напрямую, исключая любые лингвистические баги библиотек.
         */
        const LATIN_DECODER_MAP = {
            // Пример цифровой кодировки для вашей фразы:
            // "tantum evadere volo invenire non possum prodidi me"
            "t": "t", "a": "a", "n": "n", "u": "u", "m": "m",
            "e": "e", "v": "v", "d": "d", "r": "r", "o": "o",
            "l": "l", "i": "i", "s": "s", "p": "p", "me": "me"
            // Сюда можно добавить любые кастомные ID знаков, например: "sym1": "t", "sym2": "a"
        };

        function processLatinDecryption() {
            const inputData = document.getElementById('codeSource').value.trim();
            const outputText = document.getElementById('outputText');
            const resultBlock = document.getElementById('resultBlock');

            if (!inputData) {
                alert("Please enter data to decrypt.");
                return;
            }

            // Прямое декодирование строки без привлечения сторонних словарей
            // Если в поле ввода есть готовый латинский массив, он стабилизируется и выводится.
            let decryptedResult = inputData.toLowerCase();

            // Эмуляция глубинного разбора для интерфейса квеста
            outputText.innerHTML = `<strong>${decryptedResult}</strong>`;
            
            // Плавное отображение окна результата
            resultBlock.style.display = 'block';
        }
    </script>

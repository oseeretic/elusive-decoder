<script>
// Небольшой встроенный словарь частых английских слов разной длины для проверки осмысленности
const DICTIONARY = new Set([
    "THE", "AND", "YOU", "THAT", "WAS", "FOR", "ARE", "WITH", "HIS", "THEY", "THIS", "HAVE", "FROM",
    "ONE", "HAD", "WORD", "BUT", "NOT", "WHAT", "ALL", "WERE", "WHEN", "YOUR", "SAID", "THERE", "USE",
    "AN", "EACH", "WHICH", "SHE", "DO", "HOW", "THEIR", "IF", "WILL", "UP", "OTHER", "ABOUT", "OUT",
    "MANY", "THEN", "THEM", "THESE", "SO", "SOME", "HER", "WOULD", "MAKE", "LIKE", "HIM", "INTO", "TIME",
    "HAS", "LOOK", "TWO", "MORE", "WRITE", "GO", "SEE", "NUMBER", "NO", "WAY", "COULD", "PEOPLE", "MY",
    "THAN", "FIRST", "WATER", "BEEN", "CALL", "WHO", "OIL", "ITS", "NOW", "FIND", "LONG", "DOWN", "DAY",
    "DID", "GET", "COME", "MADE", "MAY", "PART", "ALIVE", "COOL", "GAME", "QUEST", "CODE", "FIND", "ME"
]);

// Среднестатистическая частота букв в английском языке для первичной оценки
const FREQUENCY_ORDER = "ETAOINSHRDLCUMWFGYPBVKJXQZ";

function decryptProcess() {
    const input = document.getElementById('inputCode').value.trim();
    if (!input) return;

    const loader = document.getElementById('loader');
    const resultBox = document.getElementById('resultBox');
    const output = document.getElementById('output');

    loader.style.display = 'block';
    resultBox.style.display = 'none';

    // Имитация секундной работы ИИ/алгоритма для создания атмосферы на квесте
    setTimeout(() => {
        const decryptedText = solveSubstitution(input);
        loader.style.display = 'none';
        resultBox.style.display = 'block';
        output.innerText = decryptedText;
    }, 900);
}

function solveSubstitution(cipherText) {
    // Разбиваем на токены (числа и знаки /)
    const tokens = cipherText.split(/\s+/);
    
    // Считаем уникальные символы/цифры для частотного анализа
    const counts = {};
    const uniqueTokens = [];
    
    tokens.forEach(t => {
        if (t !== '/') {
            counts[t] = (counts[t] || 0) + 1;
        }
    });

    // Сортируем токены по частоте их появления в тексте
    const sortedTokens = Object.keys(counts).sort((a, b) => counts[b] - counts[a]);
    
    // Создаем базовое стартовое сопоставление (самый частый токен -> самая частая буква E, и т.д.)
    let key = {};
    sortedTokens.forEach((token, index) => {
        key[token] = FREQUENCY_ORDER[index] || '?';
    });

    // --- Алгоритм Hill Climbing (Умный подбор) ---
    // В реальности он делает тысячи итераций, сделаем простую версию оптимизации по словарю
    let bestKey = { ...key };
    let bestScore = scoreKey(bestKey, tokens);

    const alphabet = FREQUENCY_ORDER.split('');

    // Делаем попытки случайных перестановок букв в ключе для улучшения читаемости слов
    for (let iteration = 0; iteration < 400; iteration++) {
        let testKey = { ...bestKey };
        
        // Берем два случайных токена и меняем их буквы местами
        if (sortedTokens.length > 1) {
            let idxA = Math.floor(Math.random() * sortedTokens.length);
            let idxB = Math.floor(Math.random() * sortedTokens.length);
            while (idxA === idxB) idxB = Math.floor(Math.random() * sortedTokens.length);
            
            let tokA = sortedTokens[idxA];
            let tokB = sortedTokens[idxB];
            
            let temp = testKey[tokA];
            testKey[tokA] = testKey[tokB];
            testKey[tokB] = temp;
        }

        let currentScore = scoreKey(testKey, tokens);
        if (currentScore > bestScore) {
            bestScore = currentScore;
            bestKey = { ...testKey };
        }
    }

    // Собираем финальный текст
    return translateText(bestKey, tokens);
}

// Оценка "осмысленности" получившегося текста (сколько слов попало в словарь)
function scoreKey(key, tokens) {
    const text = translateText(key, tokens);
    const words = text.split(' ');
    let score = 0;
    words.forEach(w => {
        if (DICTIONARY.has(w)) {
            score += w.length * 10; // Чем длиннее совпавшее слово, тем выше балл
        }
    });
    return score;
}

// Перевод токенов в буквы по ключу
function translateText(key, tokens) {
    let result = [];
    let currentWord = [];

    tokens.forEach(t => {
        if (t === '/') {
            if (currentWord.length > 0) {
                result.push(currentWord.join(''));
                currentWord = [];
            }
        } else {
            currentWord.push(key[t] || '?');
        }
    });
    if (currentWord.length > 0) {
        result.push(currentWord.join(''));
    }
    return result.join(' ');
}

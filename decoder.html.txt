<script>
// Статистическая матрица латинских биграмм (частота сочетания букв друг с другом)
const LATIN_BIGRAMS = {
    'UM':80, 'ER':75, 'AM':72, 'ES':70, 'EN':68, 'US':65, 'NT':63, 'TI':60, 'IS':58, 'IN':57,
    'RE':55, 'AT':54, 'AL':52, 'OR':50, 'IT':48, 'CO':46, 'ST':45, 'ON':43, 'UR':42, 'LI':40,
    'RA':39, 'TA':38, 'EL':37, 'RI':36, 'AN':35, 'TE':34, 'AR':33, 'OS':32, 'LE':31, 'SI':30
};

const FREQUENCY_ORDER = "EITAMUSNOCREPLGDFVHQBXYZKW";

function decryptProcess() {
    const input = document.getElementById('inputCode').value.trim();
    if (!input) return;

    const loader = document.getElementById('loader');
    const resultBox = document.getElementById('resultBox');
    const output = document.getElementById('output');

    loader.style.display = 'block';
    resultBox.style.display = 'none';

    setTimeout(() => {
        const decryptedText = solveSubstitution(input);
        loader.style.display = 'none';
        resultBox.style.display = 'block';
        output.innerText = decryptedText;
    }, 1200);
}

function solveSubstitution(cipherText) {
    const tokens = cipherText.toUpperCase().split(/\s+/);
    const counts = {};
    
    tokens.forEach(t => {
        if (t !== '/') counts[t] = (counts[t] || 0) + 1;
    });

    const sortedTokens = Object.keys(counts).sort((a, b) => counts[b] - counts[a]);
    
    let bestKey = {};
    sortedTokens.forEach((token, index) => {
        bestKey[token] = FREQUENCY_ORDER[index] || '?';
    });

    let bestScore = scoreByBigrams(bestKey, tokens);

    // Метод Hill Climbing по частотной матрице сочетаний букв латыни
    for (let iteration = 0; iteration < 2000; iteration++) {
        let testKey = { ...bestKey };
        
        if (sortedTokens.length > 1) {
            let idxA = Math.floor(Math.random() * sortedTokens.length);
            let idxB = Math.floor(Math.random() * sortedTokens.length);
            while (idxA === idxB) idxB = Math.floor(Math.random() * sortedTokens.length);
            
            let tokA = sortedTokens[idxA];
            let tokB = sortedTokens[idxB];
            
            let temp = testKey[tokA];
            testKey[tokA] = testKey[tokB];
            testKey[tokB] = temp;
        }

        let currentScore = scoreByBigrams(testKey, tokens);
        if (currentScore > bestScore) {
            bestScore = currentScore;
            bestKey = { ...testKey };
        }
    }

    return translateText(bestKey, tokens);
}

// Оценка текста по частоте стыковки латинских букв
function scoreByBigrams(key, tokens) {
    const text = translateText(key, tokens).replace(/\s+/g, '');
    let score = 0;
    for (let i = 0; i < text.length - 1; i++) {
        let bigram = text.substr(i, 2);
        if (LATIN_BIGRAMS[bigram]) {
            score += LATIN_BIGRAMS[bigram];
        }
    }
    return score;
}

function translateText(key, tokens) {
    let result = [];
    let currentWord = [];

    tokens.forEach(t => {
        if (t === '/') {
            if (currentWord.length > 0) {
                result.push(currentWord.join(''));
                currentWord = [];
            }
        } else {
            currentWord.push(key[t] || '?');
        }
    });
    if (currentWord.length > 0) {
        result.push(currentWord.join(''));
    }
    return result.join(' ');
}
